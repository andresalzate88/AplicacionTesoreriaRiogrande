-- 2. TABLAS MAESTRAS Y PARÁMETROS
-- ============================================================

CREATE TABLE sedes (
    id          uuid        PRIMARY KEY DEFAULT gen_random_uuid(),
    codigo      text        UNIQUE NOT NULL,  -- DMA / CAC / APA / QBO
    nombre      text        NOT NULL,
    letra       text        UNIQUE NOT NULL,  -- D / C / A / Q
    cuenta_caja text        NOT NULL,         -- Cuenta PUC (130501/02/03/04) — por precaución
    diario_caja text        NOT NULL,         -- Diario Odoo que identifica la caja de esta sede
    activa      boolean     NOT NULL DEFAULT true,
    created_at  timestamptz NOT NULL DEFAULT now()
);

-- Nueva en v7: NIT y razón social de cada aliado para desnormalización
CREATE TABLE aliados (
    id           uuid        PRIMARY KEY DEFAULT gen_random_uuid(),
    nombre       text        UNIQUE NOT NULL,  -- ALPINA / CARNICOS / FAMILIA / NUTRESA / MEALS
    letra        text        UNIQUE NOT NULL,  -- A / C / F / N / M
    nit          text        NOT NULL,
    razon_social text        NOT NULL,
    activo       boolean     NOT NULL DEFAULT true,
    created_at   timestamptz NOT NULL DEFAULT now()
);

CREATE TABLE operaciones (
    id          uuid    PRIMARY KEY DEFAULT gen_random_uuid(),
    codigo      text    UNIQUE NOT NULL,  -- DA, DC, DF, CA, CC, CN, AA, QC, QN, QM
    sede_id     uuid    NOT NULL REFERENCES sedes(id),
    aliado_id   uuid    NOT NULL REFERENCES aliados(id),  -- FK a aliados (antes texto libre)
    tipo_erp    text    NOT NULL,                          -- SIDIS / ECOM
    activa      boolean NOT NULL DEFAULT true
);

CREATE TABLE bancos (
    id      integer PRIMARY KEY,  -- ID numérico — se usa en Excel de ingesta
    nombre  text    UNIQUE NOT NULL,
    tipo    text    NOT NULL,     -- RIOGRANDE / EXTERNO (EXTERNO = dropdown Alpina)
    activo  boolean NOT NULL DEFAULT true
);

-- operacion_id eliminado en v7 — la distribución ya está en codigo_odoo JSON
CREATE TABLE cuentas_analiticas (
    id          uuid    PRIMARY KEY DEFAULT gen_random_uuid(),
    sede_id     uuid    NOT NULL REFERENCES sedes(id),
    codigo_odoo text    NOT NULL,  -- JSON: {"7":100} / {"7":50,"8":50}
    nombre      text    NOT NULL,  -- "DMA-Alpina 100%"
    activo      boolean NOT NULL DEFAULT true
);

-- Rediseñado en v7: tabla simple con campos opcionales por tipo
-- Los tipos son fijos (no se crean/eliminan desde UI):
--   retenciones clientes | retenciones a proveedores | gastos | impuestos en gastos |
--   consignaciones a riogrande | anticipos a aliados | traslado entre cajas |
--   anticipo de nomina | aprovechamientos | anticipo recibido clientes |
--   descuento anticipo clientes
-- Guía de parametrización:
--   id_externo_odoo  → obligatorio para: retenciones clientes, retenciones a proveedores,
--                       gastos, impuestos en gastos
--   diario_odoo      → obligatorio para: consignaciones a riogrande, traslado entre cajas
--   banco_id         → obligatorio para: consignaciones a riogrande
--   cuenta           → obligatorio para: anticipos a aliados, anticipo de nomina
CREATE TABLE parametros_contables (
    id                             uuid    PRIMARY KEY DEFAULT gen_random_uuid(),
    sede_id                        uuid    REFERENCES sedes(id),  -- NULL = todas las sedes
    tipo_asiento                   text    NOT NULL,
    detalle_asiento                text    NOT NULL,  -- nombre para dropdown UI
    cuenta                         text,              -- Cuenta PUC (obligatoria para algunos tipos)
    id_externo_odoo                text,              -- Identificador exacto en Odoo
    diario_odoo                    text,              -- Diario Odoo destino
    banco_id                       integer REFERENCES bancos(id),  -- Solo consignaciones RG
    requiere_documento_electronico boolean DEFAULT false,           -- Solo gastos
    tope_maximo                    numeric,                         -- Solo gastos
    activo                         boolean NOT NULL DEFAULT true
);

-- Nueva en v7: cuentas contables transversales (no ligadas a tipo de asiento)
CREATE TABLE parametros_contables_generales (
    id          uuid        PRIMARY KEY DEFAULT gen_random_uuid(),
    clave       text        UNIQUE NOT NULL,  -- CUENTA_POR_PAGAR / CUENTA_POR_COBRAR
    valor       text        NOT NULL,
    descripcion text,
    activo      boolean     NOT NULL DEFAULT true,
    updated_at  timestamptz NOT NULL DEFAULT now()
);

CREATE TABLE parametros_sistema (
    id          uuid        PRIMARY KEY DEFAULT gen_random_uuid(),
    clave       text        UNIQUE NOT NULL,
    -- MARGEN_DIAS_ANTES / MARGEN_DIAS_DESPUES / TIMEOUT_SESION_MINUTOS /
    -- HORA_SYNC_AUTOMATICA / MAX_DIAS_SIN_APROBAR /
    -- RUTA_SOPORTES_DMA / RUTA_SOPORTES_CAC / RUTA_SOPORTES_APA / RUTA_SOPORTES_QBO /
    -- INGESTA_MANUAL_ACTIVA / INGESTA_AUTOMATICA_ACTIVA /
    -- SYNC_ODOO_MANUAL_ACTIVA / SYNC_ODOO_AUTOMATICA_ACTIVA /
    -- DIAS_ATRAS_REGISTRO_GASTO / DIAS_ADELANTE_REGISTRO_GASTO
    valor       text        NOT NULL,
    descripcion text,
    updated_at  timestamptz NOT NULL DEFAULT now()
);

CREATE TABLE empleados (
    id                uuid        PRIMARY KEY DEFAULT gen_random_uuid(),
    sede_id           uuid        NOT NULL REFERENCES sedes(id),
    nit               text        UNIQUE NOT NULL,
    nombre_completo   text        NOT NULL,
    cargo             text        NOT NULL,  -- CONDUCTOR / AUXILIAR_RUTA / AUXILIAR_TESORERIA / ANALISTA / OTRO
    sincronizado_odoo boolean     NOT NULL DEFAULT false,
    activo            boolean     NOT NULL DEFAULT true,
    updated_at        timestamptz NOT NULL DEFAULT now()
);

CREATE TABLE vehiculos (
    id      uuid    PRIMARY KEY DEFAULT gen_random_uuid(),
    sede_id uuid    NOT NULL REFERENCES sedes(id),
    placa   text    UNIQUE NOT NULL,
    tipo    text    NOT NULL,  -- CAMION / FURGON / MOTO / OTRO
    activo  boolean NOT NULL DEFAULT true
);

CREATE TABLE consecutivos_cuadre (
    sede_id       uuid    PRIMARY KEY REFERENCES sedes(id),
    fecha         date    NOT NULL,
    ultimo_numero integer NOT NULL DEFAULT 0
);

CREATE TABLE recaudos_dia (

    auxiliar_id         uuid,
    cerrado_at          timestamptz,
    analista_id         uuid,

    nota_devolucion     text,
    created_at          timestamptz NOT NULL DEFAULT now(),

);

-- Sin cambios v7
CREATE TABLE cuadres (

-- Renombres: parametro_iva_id→parametro_impuesto_id, tarifa_iva→tipo_impuesto, iva→valor_impuesto
-- + fecha, valor_total GENERATED, 11 campos desnormalizados para Odoo (v7)
CREATE TABLE gastos (
    id                              uuid        PRIMARY KEY DEFAULT gen_random_uuid(),
    cuadre_id                       uuid        REFERENCES cuadres(id),
    recaudo_id                      uuid        NOT NULL REFERENCES recaudos_dia(id),
    origen                          text        NOT NULL,  -- PLANILLA / DESTINO_EFECTIVO
    fecha                           date        NOT NULL,  -- restringido por DIAS_ATRAS/ADELANTE_REGISTRO_GASTO
    parametro_id                    uuid        NOT NULL REFERENCES parametros_contables(id),
    parametro_impuesto_id           uuid        REFERENCES parametros_contables(id),
    parametro_retencion_id          uuid        REFERENCES parametros_contables(id),
    proveedor_id                    uuid        REFERENCES proveedores(id),
    cuenta_analitica_id             uuid        REFERENCES cuentas_analiticas(id),
    documento_electronico_proveedor text,
    -- Valores ingresados por auxiliar
    valor_base                      numeric     NOT NULL,
    valor_impuesto                  numeric     NOT NULL DEFAULT 0,
    valor_retencion                 numeric     NOT NULL DEFAULT 0,
    valor_total                     numeric     GENERATED ALWAYS AS (valor_base + valor_impuesto - valor_retencion) STORED,
    -- Desnormalizados al momento del registro
    tipo_impuesto                   text,       -- DEN: parametros_contables.detalle_asiento (impuesto)
    nit_proveedor                   text,       -- DEN: proveedores.nit
    nombre_proveedor                text,       -- DEN: proveedores.nombre
    detalle_gasto                   text,       -- DEN: parametros_contables.detalle_asiento (gasto)
    id_externo_gasto                text,       -- DEN: parametros_contables.id_externo_odoo (gasto)
    detalle_impuesto                text,       -- DEN: parametros_contables.detalle_asiento (impuesto)
    id_externo_impuesto             text,       -- DEN: parametros_contables.id_externo_odoo (impuesto)
    detalle_retencion               text,       -- DEN: parametros_contables.detalle_asiento (retención)
    id_externo_retencion            text,       -- DEN: parametros_contables.id_externo_odoo (retención)
    codigo_analitica_odoo           text,       -- DEN: cuentas_analiticas.codigo_odoo
    diario_caja                     text,       -- DEN: sedes.diario_caja
    -- Control
    supera_tope                     boolean     NOT NULL DEFAULT false,
    justificacion_tope              text,
    created_at                      timestamptz NOT NULL DEFAULT now()
);

-- -------------------------------------------------------
-- generar_consecutivo_cuadre
-- Genera: DMA-110426.01 — con FOR UPDATE para evitar duplicados
-- -------------------------------------------------------
CREATE OR REPLACE FUNCTION generar_consecutivo_cuadre(
    p_sede_id uuid,
    p_fecha   date
) RETURNS text AS $$
DECLARE
    v_codigo    text;
    v_fecha_str text;
    v_numero    integer;
BEGIN
    SELECT codigo INTO v_codigo FROM sedes WHERE id = p_sede_id;
    v_fecha_str := to_char(p_fecha, 'DDMMYY');
    UPDATE consecutivos_cuadre
    SET ultimo_numero = CASE
            WHEN fecha = p_fecha THEN ultimo_numero + 1
            ELSE 1 END,
        fecha = p_fecha
    WHERE sede_id = p_sede_id
    RETURNING ultimo_numero INTO v_numero;
    RETURN v_codigo || '-' || v_fecha_str || '.' || lpad(v_numero::text, 2, '0');
END;
$$ LANGUAGE plpgsql;

-- -------------------------------------------------------
-- generar_consecutivo_recaudo
-- Genera: DMA-RD-110426 / DMA-RD-110426-R1 si rehecho
-- -------------------------------------------------------
CREATE OR REPLACE FUNCTION generar_consecutivo_recaudo(
    p_sede_id uuid,
    p_fecha   date,
    p_rehecho boolean DEFAULT false
) RETURNS text AS $$
DECLARE
    v_codigo      text;
    v_fecha_str   text;
    v_consecutivo text;
    v_count       integer;
BEGIN
    SELECT codigo INTO v_codigo FROM sedes WHERE id = p_sede_id;
    v_fecha_str   := to_char(p_fecha, 'DDMMYY');
    v_consecutivo := v_codigo || '-RD-' || v_fecha_str;
    IF p_rehecho THEN
        SELECT COUNT(*) INTO v_count
        FROM recaudos_dia
        WHERE sede_id = p_sede_id AND fecha = p_fecha;
        v_consecutivo := v_consecutivo || '-R' || v_count::text;
    END IF;
    RETURN v_consecutivo;
END;
$$ LANGUAGE plpgsql;

-- -------------------------------------------------------
-- get_saldo_anterior
-- -------------------------------------------------------
CREATE OR REPLACE FUNCTION get_saldo_anterior(
    p_sede_id uuid,
    p_fecha   date
) RETURNS numeric AS $$
DECLARE v_saldo numeric;
BEGIN
    SELECT nuevo_saldo INTO v_saldo
    FROM recaudos_dia
    WHERE sede_id = p_sede_id
      AND fecha < p_fecha
      AND estado = 'APROBADO'
    ORDER BY fecha DESC LIMIT 1;
    RETURN COALESCE(v_saldo, 0);
END;
$$ LANGUAGE plpgsql;

-- -------------------------------------------------------
-- validar_dias_sin_aprobar
-- -------------------------------------------------------
CREATE OR REPLACE FUNCTION validar_dias_sin_aprobar(
    p_sede_id uuid
) RETURNS boolean AS $$
DECLARE
    v_max  integer;
    v_dias integer;
BEGIN
    SELECT valor::integer INTO v_max
    FROM parametros_sistema WHERE clave = 'MAX_DIAS_SIN_APROBAR';
    SELECT COUNT(*) INTO v_dias
    FROM recaudos_dia
    WHERE sede_id = p_sede_id AND estado = 'CERRADO_AUXILIAR';
    RETURN v_dias <= COALESCE(v_max, 2);
END;
$$ LANGUAGE plpgsql;

-- -------------------------------------------------------
-- liberar_bloqueos_expirados
-- Parámetro opcional p_sede_id para liberar solo una sede
-- -------------------------------------------------------
CREATE OR REPLACE FUNCTION liberar_bloqueos_expirados(
    p_sede_id uuid DEFAULT NULL
) RETURNS void AS $$
DECLARE v_timeout integer;
BEGIN
    SELECT valor::integer INTO v_timeout
    FROM parametros_sistema WHERE clave = 'TIMEOUT_SESION_MINUTOS';
    UPDATE consignaciones_banco
    SET estado_cuadre = 'LIBRE', bloqueada_por = NULL, bloqueada_at = NULL
    WHERE estado_cuadre = 'BLOQUEADA'
      AND bloqueada_at < now() - (v_timeout || ' minutes')::interval
      AND (p_sede_id IS NULL OR sede_id = p_sede_id);
    UPDATE consignaciones_aliados
    SET estado_cuadre = 'LIBRE', bloqueada_por = NULL, bloqueada_at = NULL
    WHERE estado_cuadre = 'BLOQUEADA'
      AND bloqueada_at < now() - (v_timeout || ' minutes')::interval
      AND (p_sede_id IS NULL OR sede_id = p_sede_id);
END;
$$ LANGUAGE plpgsql;

-- -------------------------------------------------------
-- cruzar_documentos_dian
-- -------------------------------------------------------
CREATE OR REPLACE FUNCTION cruzar_documentos_dian()
RETURNS void AS $$
BEGIN
    UPDATE documentos_erp de
    SET estado_dian      = dd.estado_dian,
        validado_dian_at = now()
    FROM documentos_dian dd
    WHERE de.documento_electronico = dd.documento_electronico
      AND de.documento_electronico IS NOT NULL;
END;
$$ LANGUAGE plpgsql;

    ON CONFLICT DO NOTHING;

    -- clientes involucrados en este recaudo
    INSERT INTO hist_clientes (
        id, operacion_codigo_cliente, operacion_id,
        numero_identificacion, digito_verificacion,
        tipo_tercero, tipo_identificacion,
        nombre_establecimiento, nombre_completo,
        nombre1, nombre2, apellido1, apellido2,
        pais, departamento, municipio, zip,
        direccion, correo, telefono,
        sincronizado_odoo, sincronizado_at,
        requiere_sync_odoo, error_sync,
        activo, updated_at,
        promovido_at, estado_odoo
    )
    SELECT
        c.id, c.operacion_codigo_cliente, c.operacion_id,
        c.numero_identificacion, c.digito_verificacion,
        c.tipo_tercero, c.tipo_identificacion,
        c.nombre_establecimiento, c.nombre_completo,
        c.nombre1, c.nombre2, c.apellido1, c.apellido2,
        c.pais, c.departamento, c.municipio, c.zip,
        c.direccion, c.correo, c.telefono,
        c.sincronizado_odoo, c.sincronizado_at,
        c.requiere_sync_odoo, c.error_sync,
        c.activo, c.updated_at,
        now(), 'PENDIENTE'
    FROM clientes c
    WHERE c.id IN (
        SELECT DISTINCT cliente_id FROM documentos_erp
        WHERE recaudo_id = p_recaudo_id AND cliente_id IS NOT NULL
    )
    ON CONFLICT DO NOTHING;

    -- proveedores involucrados en este recaudo
    INSERT INTO hist_proveedores (
        id, nit, digito_verificacion, nombre,
        tipo_identificacion, tipo_tercero,
        sincronizado_odoo, sincronizado_at,
        requiere_sync_odoo, error_sync,
        creado_por, activo, updated_at,
        promovido_at, estado_odoo
    )
    SELECT
        p.id, p.nit, p.digito_verificacion, p.nombre,
        p.tipo_identificacion, p.tipo_tercero,
        p.sincronizado_odoo, p.sincronizado_at,
        p.requiere_sync_odoo, p.error_sync,
        p.creado_por, p.activo, p.updated_at,
        now(), 'PENDIENTE'
    FROM proveedores p
    WHERE p.id IN (
        SELECT DISTINCT proveedor_id FROM gastos
        WHERE recaudo_id = p_recaudo_id AND proveedor_id IS NOT NULL
    )
    ON CONFLICT DO NOTHING;

    -- empleados involucrados en anticipos de este recaudo
    INSERT INTO hist_empleados (
        id, sede_id, nit, nombre_completo, cargo,
        sincronizado_odoo, activo, updated_at,
        promovido_at, estado_odoo
    )
    SELECT
        e.id, e.sede_id, e.nit, e.nombre_completo, e.cargo,
        e.sincronizado_odoo, e.activo, e.updated_at,
        now(), 'PENDIENTE'
    FROM empleados e
    WHERE e.id IN (
        SELECT DISTINCT empleado_id FROM cuadre_anticipos_nomina
        WHERE recaudo_id = p_recaudo_id
    )
    ON CONFLICT DO NOTHING;

END;
$$ LANGUAGE plpgsql;

-- ── Proteger históricas y audit_log ─────────────────────────
CREATE OR REPLACE FUNCTION fn_proteger_historico() RETURNS trigger AS $$
BEGIN
    RAISE EXCEPTION 'La tabla % es inmutable. No se permite % en ella.', TG_TABLE_NAME, TG_OP;
    RETURN NULL;
END;
$$ LANGUAGE plpgsql;

DO $$
DECLARE t text;
DECLARE tablas_hist text[] := ARRAY[
    'hist_recaudos_dia','hist_cuadres','hist_retenciones',
    'hist_gastos','hist_anticipos_nomina','hist_traslados_caja',
    'hist_soportes_dia','hist_checklist_revision','hist_documentos_erp',
    'hist_consignaciones_banco','hist_consignaciones_aliados',
    'hist_clientes','hist_proveedores','hist_empleados','hist_documentos_dian',
    'audit_log'
];
BEGIN
    FOREACH t IN ARRAY tablas_hist LOOP
        EXECUTE format('
            CREATE TRIGGER trg_proteger_%I
            BEFORE UPDATE OR DELETE ON %I
            FOR EACH ROW EXECUTE FUNCTION fn_proteger_historico()
        ', t, t);
    END LOOP;
END;
$$;

-- ── Cruce automático DIAN ────────────────────────────────────
CREATE OR REPLACE FUNCTION fn_cruzar_dian_on_insert() RETURNS trigger AS $$
BEGIN
    UPDATE documentos_erp
    SET estado_dian      = NEW.estado_dian,
        validado_dian_at = now()
    WHERE documento_electronico = NEW.documento_electronico
      AND documento_electronico IS NOT NULL;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_cruzar_dian
    AFTER INSERT OR UPDATE ON documentos_dian
    FOR EACH ROW EXECUTE FUNCTION fn_cruzar_dian_on_insert();

-- ── Marcar requiere_sync_odoo cuando cambian datos del cliente
CREATE OR REPLACE FUNCTION fn_cliente_changed() RETURNS trigger AS $$
BEGIN
    IF OLD.sincronizado_odoo = true AND (
        OLD.numero_identificacion  IS DISTINCT FROM NEW.numero_identificacion  OR
        OLD.nombre_establecimiento IS DISTINCT FROM NEW.nombre_establecimiento OR
        OLD.nombre_completo        IS DISTINCT FROM NEW.nombre_completo        OR
        OLD.correo                 IS DISTINCT FROM NEW.correo                 OR
        OLD.telefono               IS DISTINCT FROM NEW.telefono               OR
        OLD.direccion              IS DISTINCT FROM NEW.direccion
    ) THEN
        NEW.requiere_sync_odoo := true;
        NEW.sincronizado_odoo  := false;
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_cliente_changed
    BEFORE UPDATE ON clientes
    FOR EACH ROW EXECUTE FUNCTION fn_cliente_changed();

-- ── Marcar requiere_sync_odoo cuando cambian datos del proveedor
CREATE OR REPLACE FUNCTION fn_proveedor_changed() RETURNS trigger AS $$
BEGIN
    IF OLD.sincronizado_odoo = true AND (
        OLD.nit    IS DISTINCT FROM NEW.nit    OR
        OLD.nombre IS DISTINCT FROM NEW.nombre
    ) THEN
        NEW.requiere_sync_odoo := true;
        NEW.sincronizado_odoo  := false;
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_proveedor_changed
    BEFORE UPDATE ON proveedores
    FOR EACH ROW EXECUTE FUNCTION fn_proveedor_changed();

-- ── Audit log — protegido con NULLIF en cast uuid ────────────
CREATE OR REPLACE FUNCTION fn_audit_log() RETURNS trigger AS $$
BEGIN
    INSERT INTO audit_log (
        tabla, registro_id, accion,
        usuario_id, usuario_email, ip_address,
        valor_anterior, valor_nuevo
    ) VALUES (
        TG_TABLE_NAME,
        COALESCE(NEW.id, OLD.id),
        TG_OP,
        NULLIF(current_setting('app.current_user_id',    true), '')::uuid,
        current_setting('app.current_user_email', true),
        current_setting('app.current_ip',         true),
        CASE WHEN TG_OP IN ('UPDATE','DELETE') THEN to_jsonb(OLD) ELSE NULL END,
        CASE WHEN TG_OP IN ('INSERT','UPDATE') THEN to_jsonb(NEW) ELSE NULL END
    );
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DECLARE tablas_audit text[] := ARRAY[
    'sedes','aliados','operaciones','bancos','cuentas_analiticas',

    'empleados','vehiculos',
    'clientes','proveedores',
    'cuadres','cuadre_retenciones','gastos',
    'cuadre_anticipos_nomina','traslados_caja',
    'recaudos_dia','checklist_revision'
];
BEGIN
    FOREACH t IN ARRAY tablas_audit LOOP
        EXECUTE format('
            CREATE TRIGGER trg_audit_%I
            AFTER INSERT OR UPDATE OR DELETE ON %I
